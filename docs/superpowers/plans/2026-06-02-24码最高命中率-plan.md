#!/usr/bin/env python3
"""49选24策略回测优化器"""

import sys
import os
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from data_fetcher import get_all_records, build_standard_records
from predictors.active.number_selection_v1.predictor import NumberSelectionPredictorV1

def backtest_strategy(strategy, lookback, top_k=24, start_idx=50):
    """回测指定策略"""
    records = get_all_records([2024, 2025, 2026])
    standard = build_standard_records(records)

    predictor = NumberSelectionPredictorV1()

    hits = 0
    total = 0

    for i in range(start_idx, len(standard)):
        history = standard[:i]
        try:
            prediction = predictor.predict(history, top_k=top_k, strategy=strategy, lookback=lookback)
            actual = standard[i].get('特码号码')
            if actual in prediction:
                hits += 1
            total += 1
        except Exception as e:
            print(f"Error at {i}: {e}")
            continue

    return hits, total, hits/total*100 if total > 0 else 0

if __name__ == "__main__":
    # 策略列表
    strategies = [
        ("freq_only", [50, 100, 0]),  # lookback=0表示全部
        ("gap_freq", [50, 100]),
        ("multi_window", [[20, 50, 100], "union"]),
        ("multi_window", [[20, 50, 100], "intersection"]),
    ]

    print("策略回测结果:")
    print("=" * 60)
    for strategy_name, params in strategies:
        hits, total, rate = backtest_strategy(strategy_name, params)
        print(f"{strategy_name} {params}: {hits}/{total} = {rate:.2f}%")
```

def _compute_multi_window_scores(self, history, windows=[20, 50, 100], mode="union"):
    """
    多窗口频率融合

    windows: 滚动窗口列表
    mode: "union"(并集) 或 "intersection"(交集)
    """
    from collections import Counter

    all_scores = {}
    for w in windows:
        recent = history[-w:] if w > 0 else history
        nums = [r.get('特码号码') for r in recent if r.get('特码号码')]
        freq = Counter(nums)

        for num, count in freq.items():
            if num not in all_scores:
                all_scores[num] = 0
            all_scores[num] += count

    # 并集：所有窗口分数相加
    # 交集：取平均
    if mode == "intersection":
        for num in all_scores:
            all_scores[num] /= len(windows)

    return all_scores
```
